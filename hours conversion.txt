#include <iostream>
using namespace std;
 int main () {
     
     int hour = 1;
     
     int min = 60;
     int sec = min * 60;
     int milisec = sec * 1000;
     
     cout << "hours:" << hour << "\n";
     cout << "minutes:" << min << "\n";
     cout << "seconds:" << sec << "\n";
     cout << "Miliseconds:" << milisec << "\n";
     
       return 0;
 }
